        document.addEventListener('DOMContentLoaded', () => {
            const mainTabs = document.querySelectorAll('.main-tab-btn');
            const myTabs = document.querySelectorAll('.my-tab-btn');
            const myContentTabs = document.querySelectorAll('.my-content-tab');

            // Set initial active states for myTabs
            myTabs.forEach(t => {
                const tabName = t.getAttribute('data-content');
                if (tabName === 'reservation') {
                    t.classList.remove('text-brown-900');
                    t.classList.add('bg-hazelnut', 'text-white');
                } else {
                    t.classList.remove('bg-hazelnut', 'text-white');
                    t.classList.add('bg-white', 'text-brown-900', 'hover:bg-gray-100');
                }
            });

            // My Jarimae Tabs Logic
            myTabs.forEach(tab => {
                tab.addEventListener('click', () => {
                    myTabs.forEach(t => {
                        t.classList.remove('bg-hazelnut', 'text-white');
                        t.classList.add('bg-white', 'text-brown-900', 'hover:bg-gray-100');
                    });
                    
                    const tabName = tab.getAttribute('data-content');
                    tab.classList.remove('bg-white', 'text-brown-900', 'hover:bg-gray-100');
                    if (tabName === 'reservation') {
                         tab.classList.add('bg-hazelnut', 'text-white');
                    } else if (tabName === 'delivery') {
                         tab.classList.add('bg-white', 'text-brown-900');
                    } else if (tabName === 'waiting') {
                         tab.classList.add('bg-white', 'text-brown-900');
                    }

                    myContentTabs.forEach(content => content.classList.add('hidden'));
                    const targetContentId = tab.getAttribute('data-content') + '-content';
                    document.getElementById(targetContentId).classList.remove('hidden');
                });
            });

            // Calendar Logic
            const calendarDatesEl = document.getElementById('calendar-dates');
            const currentMonthEl = document.getElementById('current-month');
            const prevMonthBtn = document.getElementById('prev-month-btn');
            const nextMonthBtn = document.getElementById('next-month-btn');

            let currentDate = new Date();
            const events = {
                '2025-08-28': ['예약', '배달'],
                '2025-09-05': ['예약'],
                '2025-09-10': ['예약'],
                '2025-09-15': ['웨이팅'],
            };

            const renderCalendar = (date) => {
                calendarDatesEl.innerHTML = '';
                const year = date.getFullYear();
                const month = date.getMonth();
                const firstDay = new Date(year, month, 1).getDay();
                const lastDate = new Date(year, month + 1, 0).getDate();

                currentMonthEl.textContent = `${year}년 ${month + 1}월`;

                for (let i = 0; i < firstDay; i++) {
                    calendarDatesEl.appendChild(document.createElement('div'));
                }

                for (let i = 1; i <= lastDate; i++) {
                    const cell = document.createElement('div');
                    cell.textContent = i;
                    cell.classList.add('date-cell', 'text-brown-900', 'p-1', 'flex', 'items-center', 'justify-center', 'aspect-square');

                    const dateString = `${year}-${String(month + 1).padStart(2, '0')}-${String(i).padStart(2, '0')}`;
                    if (events[dateString]) {
                        cell.classList.add('event-day');
                    }
                    calendarDatesEl.appendChild(cell);
                }
            };

            prevMonthBtn.addEventListener('click', () => {
                currentDate.setMonth(currentDate.getMonth() - 1);
                renderCalendar(currentDate);
            });

            nextMonthBtn.addEventListener('click', () => {
                currentDate.setMonth(currentDate.getMonth() + 1);
                renderCalendar(currentDate);
            });

            renderCalendar(currentDate);

             // Main Tabs Logic
            mainTabs.forEach(tab => {
                tab.addEventListener('click', () => {
                    mainTabs.forEach(t => {
                        t.classList.remove('bg-hazelnut', 'text-white');
                        t.classList.add('bg-white', 'text-brown-900', 'hover:bg-gray-100');
                    });
                    tab.classList.remove('bg-white', 'text-brown-900', 'hover:bg-gray-100');
                    tab.classList.add('bg-hazelnut', 'text-white');
                    const tabName = tab.getAttribute('data-tab');
                    if (tabName === '예약' || tabName === '웨이팅') {
                        searchInput.placeholder = '식당명, 메뉴 검색';
                    } else if (tabName === '배달') {
                        searchInput.placeholder = '식당명, 음식 종류 검색';
                    }
                });
            });
        });
    </script>
